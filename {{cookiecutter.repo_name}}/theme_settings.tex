   \newcounter{framenumberappendix}
   \setcounter{framenumberappendix}{\value{framenumber}}
}

   \addtocounter{framenumberappendix}{-\value{framenumber}}
   \addtocounter{framenumber}{\value{framenumberappendix}}
}
